        function toggleFullScreen() {

            const doc = document.documentElement;

            const btn = document.getElementById('fs-button');

            const toast = document.getElementById('toast');


            if (!document.fullscreenElement) {

                // Intento de pantalla completa real

                doc.requestFullscreen().then(() => {

                    toast.innerText = "Modo Pantalla Completa Activado.";

                    showToast();

                }).catch(err => {

                    // FALLBACK: Modo Enfoque Visual si la API está bloqueada

                    document.body.classList.add('presentation-mode');

                    toast.innerText = "Modo Enfoque Activado (Navegador bloqueó Fullscreen real).";

                    showToast();

                    btn.innerHTML = '<i class="fas fa-compress"></i> Salir Enfoque';

                });

            } else {

                if (document.exitFullscreen) {

                    document.exitFullscreen();

                    document.body.classList.remove('presentation-mode');

                    btn.innerHTML = '<i class="fas fa-expand"></i> Presentar';

                }

            }

        }


        // Listener para detectar salida de fullscreen nativo

        document.addEventListener('fullscreenchange', () => {

            if (!document.fullscreenElement) {

                document.body.classList.remove('presentation-mode');

                document.getElementById('fs-button').innerHTML = '<i class="fas fa-expand"></i> Presentar';

            }

        });


        function showToast() {

            const x = document.getElementById("toast");

            x.className = "show";

            setTimeout(function(){ x.className = x.className.replace("show", ""); }, 5000);

        }
